						Omega		& $ f(n) = \Omega(g(n)) $ 	& $ f(n) \geq c.g(n) $ \\
						
						Little O		& $ f(n) = o(g(n)) $		& $ f(n) < c.g(n) $ \\
						
						Little Omega	& $ f(n) = \omega(g(n)) $	& $ f(n) > c.g(n) $ \\
